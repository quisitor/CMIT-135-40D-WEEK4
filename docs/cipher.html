23   if __name__ == '__main__':                        # added to protect from sphinx
24       # Get the message to encrypt from the user
25       print('Enter your message:')
26       unencryptedMessage = input()
27   
28       # Get the encryption key from the user.  Note the error handling to make sure we have a number from 1-26
29       key = 0
30       while key < 1 or key > 26:
31           print('Enter the key number (1-26)')
32           key = int(input())
33   
34       print('The key is ' + str(key))
35   
36       # We will start with an empty string as our encryptedMessage
37       encryptedMessage = ''
38   
39       # For each symbol in the unencryptedMessage we will add an encrypted symbol into the encryptedMessage
40       for symbol in unencryptedMessage:
41           if symbol.isalpha():
42               num = ord(symbol)
43               num += key
44   
45               if symbol.isupper():
46                   if num > ord('Z'):
47                       num -= 26
48                   elif num < ord('A'):
49                       num += 26
50               elif symbol.islower():
51                   if num > ord('z'):
52                       num -= 26
53                   elif num < ord('a'):
54                       num += 26
55   
56               encryptedMessage += chr(num)
57           else:
58               encryptedMessage += symbol
59   
60       print("Your encrypted message is:")
61       print(encryptedMessage)
