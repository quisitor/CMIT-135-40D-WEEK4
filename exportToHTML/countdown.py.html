23   # Helper functions
24   
25   
26   def while_countdown(starting_number):
27       """ 
28       This function takes a number, validates its a positive integer, and prints 
29       the countdown to 0 to the terminal using a "while" loop. 
30    
31       :param starting_number: Integer > 0 
32       :terminal output: Label indicating countdown method used and a printout of the countdown 
33       """
34       print(f"Counting down from {starting_number} to 0 using a 'while loop'")
35       while starting_number >= 0:
36           print(starting_number)
37           starting_number -= 1
38   
39   
40   def for_countdown(starting_number):
41       """ 
42       This function takes a number, validates its a positive integer, and prints 
43       the countdown to 0 to the terminal using a "for" loop. 
44    
45       :param starting_number: Integer > 0 
46       :terminal output: Label indicating countdown method used and a printout of the countdown 
47       """
48       print(f"Counting down from {starting_number} to 0 using a 'for loop'")
49       for number in range(starting_number, -1, -1):
50           print(number)
51   
52   # Main
53   
54   
55   if __name__ == '__main__':
56       while_countdown(10)
57       for_countdown(10)
