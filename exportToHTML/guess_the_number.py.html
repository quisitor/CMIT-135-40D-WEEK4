27   # Imports
28   from random import randint
29   
30   # Helper Functions
31   
32   
33   def convert_str_to_num(str_number):
34       """ 
35       This function takes a string, validates it can be converted to a number 
36       and returns either a string or float representation of the string. 
37    
38       :param str_number: a string representation of a number 
39                          ex: "123", "123.3", "-123", "-123.3" 
40    
41       :return: an int or float representing the number in the string 
42    
43       """
44   
45       while True:
46           if str_number.replace('-', '', 1).isnumeric():
47               return int(str_number)
48           elif str_number.replace('-', '', 1).replace('.', '', 1).isnumeric():
49               return float(str_number)
50           else:
51               str_number = input('Please enter a valid number: ')
52   
53   
54   def lower_higher(answer, guess):
55       """ 
56       This function takes a known number and an unknown number and compares the 
57       difference. The function will return a message indicating to the calling 
58       function whether or not the user needs to guess higher or lower than the 
59       number just guessed. 
60    
61       :param answer: known integer value determined by the calling function 
62       :param guess: number passed in by the end user 
63       :return: 'Guess Higher', 'Guess Lower', True 
64    
65       """
66   
67       if answer == guess:
68           return True
69       elif guess < answer:
70           return 'Guess Higher'
71       elif guess > answer:
72           return 'Guess Lower'
73   
74   
75   # Main
76   if __name__ == '__main__':
77       random_number = randint(0, 100)
78       matching_value = False
79   
80       guessed_input = input('The computer has chosen a number at random between 0 and 100.\n '
81                             'Please try and guess the chosen number: ')
82   
83       while not matching_value:
84           guessed_number = convert_str_to_num(guessed_input)
85           result = lower_higher(random_number, guessed_number)
86   
87           if result == 'Guess Higher':
88               guessed_input = input('Guess a Higher Number: ')
89           elif result == 'Guess Lower':
90               guessed_input = input('Guess a Lower Number: ')
91           else:
92               matching_value = result
93   
94       print('Great job! You guessed the correct number: ', random_number)
